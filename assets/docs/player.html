¶

playSound stop the currently playing sound and starts the new one.

 
  playSound: (src) ->
            
    do @sound?.stop
      
    newSound = @getSound src
    if @sound is newSound  
      @sound = null
    else
      @sound = newSound
      do @sound.play
      $("*[href=\"#{src}\"]").addClass 'playing'

    return @sound
